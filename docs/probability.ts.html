export function Probability<T>(options: IProbabilityOptions<T>): T {
    let result = generate<T>(options.conditions);
    return result;
}

/**
 * Description placeholder

 * @param {IProbability<T>[]} conditions
 * @returns {T}
 */
export function generate<T>(conditions: IProbability<T>[]): T {
    // generate a new float to dictate the probability
    let previousProb = 0;
    for (const condition of conditions) {
        const probability = rngFloat({min: previousProb, max: condition.probability});
        console.log(probability);
        if (condition.generator) {
            if (condition.generator instanceof Function) {
                return condition.generator();
            } else {
                return condition.generator;
            }
        }
    }

    throw new Error('Could not generate a conditional value.');
}
