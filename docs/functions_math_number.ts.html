 * @returns {number}
 */
export function randn_bm(seed: number) {
    const u = 1 - mulberry32(seed);
    const v = mulberry32(seed);
    return Math.sqrt(-2.0 * Math.log(u)) * Math.cos(2.0 * Math.PI * v);
}

/**
 * Description placeholder
 * @date 12/9/2022 - 10:00:53 PM

 * @export
 * @param {number} a
 * @returns {number}
 */
export function mulberry32(a: number) {
    let t = a += 0x6D2B79F5;
    t = Math.imul(t ^ t >>> 15, t | 1);
    t ^= t + Math.imul(t ^ t >>> 7, t | 61);
    const result =  ((t ^ t >>> 14) >>> 0) / 4294967296;
    return result;
}

/**

 * @export
 * @param {?{min: number, max: number}} [options]
 * @returns {number}
 */
export function rngFloat(options?: {min: number, max: number}): number {
    if(!options) {
        return mulberry32(Config.seed)
    }
    Config.seed += 1;
    return mulberry32(Config.seed) * (options.max - options.min) + options.min;
}

 */
export function rngInt(options?: {min: number, max: number}): number {
    if(!options) {
        return mulberry32(Config.seed)
    }

    Config.seed += 1;
    return Math.floor(mulberry32(Config.seed) * (options.max - options.min + 1) + options.min)
}
